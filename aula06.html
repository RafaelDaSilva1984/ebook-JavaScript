Aula Operadores Condicionais


let idade = 11
if(idade >= 18 ){
console.log(`Maior de Idade = ${idade} anos`)
}else if(idade >= 12){
console.log(`Adolescente = ${idade} anos`)
}else{
console.log(`Criança = ${idade} anos`)
}

console.log('Operador ternário')
console.log(idade >= 18? 'Maior' : 'Menor')

console.log('Switch')

let genero = ''
switch(genero.toUpperCase()){
case 'M':
console.log('Masculino')
break
case 'F':
console.log('Feminino')
break
default:
console.log('Outros')
break }
}
