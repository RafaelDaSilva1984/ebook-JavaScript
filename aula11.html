// Array reduce - não altera valor original, deve iniciar com uma function, e ter o ponto de inicio em 0 zero padrão
// memo = primeiro valor nesse caso = 0 'inicial, o current, é o valor da posição de index 0 exemplo: 1
const arr = [1,4,5,5,7,9]
const result = arr.reduce(function(memo, current, index){
console.log('index:', index)
return memo + current
},0)
console.log('O resultado da soma de valores de array é: ', result)
